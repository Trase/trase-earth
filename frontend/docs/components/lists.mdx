import { PropsTable, Playground } from 'docz'
import { State } from 'react-powerplug';
import GridList from 'react-components/shared/grid-list/grid-list.component';
import GridListItem from 'react-components/shared/grid-list-item/grid-list-item.component';
import 'styles/_docz.scss';
import countries from './mock-data/countries-list.json';
import indicators from './mock-data/indicators-list.json';

# Grid List

<div className="components-container" style={{ width: 760 }}>
  <State initial={{ active: [] }}>
    {({ state, setState }) => (
      <GridList
          width={760}
          height={200}
          rowHeight={50}
          columnWidth={190}
          columnCount={4}
          items={countries}
        >
          {itemProps => (
            <GridListItem
              {...itemProps}
              isActive={state.active.includes(itemProps.item && itemProps.item.id)}
              enableItem={() => setState({ active: [...state.active, itemProps.item.id] })}
              disableItem={() => setState({ active: state.active.filter(i => i !== itemProps.item.id) })}
            />
          )}
        </GridList>
    )}
  </State>
</div>

### Items with tooltips
<div className="components-container" style={{ width: 760 }}>
  <State initial={{ active: [], activeInfo: null }}>
      {({ state, setState }) => (
        <GridList
            width={760}
            height={150}
            rowHeight={50}
            columnWidth={190}
            columnCount={4}
            items={countries.slice(0, 16)}
          >
            {itemProps => (
              <GridListItem
                {...itemProps}
                tooltip={`Lorem ipsum dolor sit ${itemProps.item.id}`}
                isInfoActive={state.activeInfo === itemProps.item.id}
                onInfoClick={it => setState({ activeInfo: state.activeInfo === it.id ? null : it.id })}
                isActive={state.active.includes(itemProps.item && itemProps.item.id)}
                enableItem={() => setState({ active: [...state.active, itemProps.item.id] })}
                disableItem={() => setState({ active: state.active.filter(i => i !== itemProps.item.id) })}
              />
            )}
          </GridList>
      )}
    </State>
</div>
